[SHOULD BE DONE IN KALI LINUX FOR CLIENT SETUP AND UBUNTU FOR RSYSLOG SERVER]
#In this case the client should be on Kali Linux
# Install rsyslog on client
sudo apt-get update
sudo apt-get install rsyslog

# Open configuration file
sudo nano /etc/rsyslog.conf

# (Add server details at end of file [Put the IP of your ubuntu server])
*.* @<ubuntu_server_ip>:514
*.* @@<ubuntu_server_ip>:514

# (Optional: add queue handling)
$ActionQueueFileName queue
$ActionQueueMaxDiskSpace 1g
$ActionQueueSaveOnShutdown on
$ActionQueueType LinkedList
$ActionResumeRetryCount -1

# Restart rsyslog service
sudo systemctl restart rsyslog

# Verify logs on server
ls /var/log/

# Monitor logs from client machine
sudo tail -f /var/log/<client-hostname>/rsyslogd.log
